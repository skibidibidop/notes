* USEFUL THINGS
	some operators
		+, -, *, /, abs, add1, ceiling, denominator, exact->inexact,
		expt, floor, gcd, log, max, numerator, quotient, random,
		remainder, sqr, tan

	string-append
		Operator
		(string-append "hi" "there") // output: hithere
		(string-append "hi " "there") // output: hi there
		(string-append "hi" " " "there") // output: hi there

	string-length
		(string-length "hello world") // output: 11
		(+ (string-length "hello world") 20) // output: 31

	-> or <-
		(string->number "30") // output: 30
		(number->string 30) // output: "30"
		(string->number "hello!") // output: #false

	and, or, not
		(and #true #true) // true
		(and #true #false) // false
		(or #true #false) // true
		(not #true) // false

	=, =?
		check if equal, return boolean

	creating simple images
		(require 2htdp/images) // include library / teachpack
		(circle 10 "solid" "red")

		(overlay (circle 5 "solid" "red")
			 (rectangle 10 "solid" "blue"))

		(empty-scene Width Height)
			Creates an empty rectangle (a scene) where other images
			can be placed

		(place-image Image Coordinates)
			Used to place image in a scene

		(place-image (circle 10 "solid" "red)
				50 80
				(empty-scene 100 100))

		0, 0 starts from the upper-left corner of the scene
		y-coordinate is measured downward

	define
		Can be used to define functions and constants

	(define <name> <expression>)

	(define (FunctionName InputName) BodyExpression)
	This is a function definition

		Usage:

		(define (y x) (* x x))
		(y 20) // This is a function application
		
		invoke function y with input 20
		output is the product of 20 * 20

		(define (moving_dot height)
			(place-image (circle 5 "solid" "red)
					47 height
					(empty-scene 100 100)))

	animate
	(require 2htdp/universe)

		Usage:

		(animate <argument>)

		cond
		Conditional statement

		(cond
			[(<condition expression 1>) <result>]
			[(<condition expression 2>) <result>]
			[(<condition expression 3>) <result>])

		Produces an error if all conditions evaluate to #false

		Usage (example in a function definition):

		(define (sign x)
			(cond
				[(> x 0) 1]
				[(= x 0) 0]
				[(< x 0) -1]))
